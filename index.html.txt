    <script>
        function updateLabel() {
            // Récupération et nettoyage des valeurs
            const zone = document.getElementById('zoneNum').value.trim();
            const allee = document.getElementById('alleeNum').value.trim();
            const etagere = document.getElementById('etagereChar').value.trim().toUpperCase();
            const etage = document.getElementById('etageNum').value.trim();
            const emplacement = document.getElementById('emplacementChar').value.trim().toUpperCase();
            const op = document.getElementById('operator').value.trim();

            // Construction du code complet (ex: C1A)
            const fullCode = etagere + etage + emplacement;

            // Mise à jour du texte
            document.getElementById('displayZone').innerText = zone;
            document.getElementById('displayAllee').innerText = allee;
            document.getElementById('displayCode').innerText = fullCode;
            document.getElementById('displayOp').innerText = op;

            // Génération du Data Matrix
            try {
                // IMPORTANT : Échelle augmentée à 5 pour l'écran pour qu'il remplisse l'espace
                bwipjs.toCanvas('barcodeCanvas', {
                    bcid: 'datamatrix',       
                    text: fullCode,           
                    scale: 5,                 
                    includetext: false,       
                });
            } catch (e) {
                console.error(e);
                alert("Erreur lors de la génération du Data Matrix. Vérifiez vos saisies.");
            }
        }

        // Initialisation automatique au chargement
        window.onload = updateLabel;
    </script>
